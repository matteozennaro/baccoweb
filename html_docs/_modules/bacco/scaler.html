    def __init__(self, OriginalCosmology, MinRadius=0.3, MaxRadius=10., mode='backward', verbose=False):
        """
        :param mode: - 'forward': we find what target redshift more closely matches a given cosmology and z
                     - 'backward': we find what redshift
        """

        if mode not in ['forward', 'backward']:
            raise ValueError("mode %s not recognised" % mode)
        if OriginalCosmology.__class__.__name__ != "Cosmology":
            raise ValueError("Original cosmology is not of class 'Cosmology'")
        self.OriginalCosmology = OriginalCosmology

        self.verbose = verbose
        self.mode = mode
        self.MinRadius = MinRadius
        self.MaxRadius = MaxRadius
        self.MinMass = OriginalCosmology.radius_to_mass(MinRadius)
        self.MaxMass = OriginalCosmology.radius_to_mass(MaxRadius)

        if self.verbose:
            logger.info("Initialising scaler for the '{0}' cosmology".format(
                self.OriginalCosmology.tag))

    def __str__(self):
        description = (("Scaler for the '{0}' cosmology \n" +
                        "\t Radius range for fitting [Mpc/h]: \t {1} - {2}\n" +
                        "\t Mass range for fitting [Msun/h]: \t {3:.2f} - {4:.2f}\n" +
                        "\t Scaling mode: \t\t  {5}" +
                        "").format(self.OriginalCosmology.tag, self.MinRadius, self.MaxRadius, np.log10(self.MinMass)[0], np.log10(self.MaxMass)[0], self.mode))
        return description

    def __del__(self):
        if self.verbose: logger.info("Successfull deletion of scaler")

    def compute_sigmadiff(self, scaleTime, scaleLength, tcosmo, radius):

        if self.mode == 'forward':
            log_targetVariance = tcosmo.get_logvariance_z(
                radius, self.OriginalCosmology.expfactor * scaleTime, cold=True)
            log_scaledVariance = self.OriginalCosmology.get_logvariance_z(
                radius/scaleLength, self.OriginalCosmology.expfactor, cold=True)
        else:
            log_targetVariance = tcosmo.get_logvariance_z(radius, tcosmo.expfactor, cold=True)
            log_scaledVariance = self.OriginalCosmology.get_logvariance_z(
                radius/scaleLength, tcosmo.expfactor*scaleTime, cold=True)

        diff = np.exp(log_targetVariance - log_scaledVariance) - 1

        return np.mean(np.abs(diff))

    def compute_growthdiff(self, scaleTime, scaleLength, tcosmo):

        # expfactor = np.linspace(0.1, 2, num=20)
        expfactor = np.linspace(0.1, 1, num=20)

        if self.mode == 'forward':
            ktarget_cosmo = None if tcosmo.pars['neutrino_mass'] == 0.0 else 0.8
            koriginal_cosmo = None if self.OriginalCosmology.pars['neutrino_mass'] == 0.0 else 0.8
            aa_fac = scaleTime/self.OriginalCosmology.expfactor
            target_growth = tcosmo.get_growth_z(expfactor, k=ktarget_cosmo)
            scaled_growth = (self.OriginalCosmology.get_growth_z(expfactor*aa_fac, k=koriginal_cosmo)
                             / self.OriginalCosmology.get_growth_z(aa_fac, k=koriginal_cosmo))
        else:
            ktarget_cosmo = None if tcosmo.pars['neutrino_mass'] == 0.0 else 0.8
            koriginal_cosmo = None if self.OriginalCosmology.pars['neutrino_mass'] == 0.0 else 0.8
            aa_fac = scaleTime/self.OriginalCosmology.expfactor
            target_growth = tcosmo.get_growth_z(expfactor, k=ktarget_cosmo)
            scaled_growth = (self.OriginalCosmology.get_growth_z(expfactor*aa_fac, k=koriginal_cosmo)
                             / self.OriginalCosmology.get_growth_z(aa_fac, k=koriginal_cosmo))

        diff = target_growth/scaled_growth - 1

        fitIndex = ((expfactor > 0.1) & (expfactor < 1))
        error_growth = np.mean(np.abs(diff[fitIndex]))

        return error_growth

[docs]    def pk_correction(self, run, wavemode=None, correction_type='additive', cold=False, nlin = False):
        '''
        Compute large-scale correction to be applied to the rescaled
        power spectrum.

        if correction_type is additive:
            returns pk_correction such that
            pk_corrected = pk_target + pk_correction

        if correction_type is multiplicative:
            retruns pk_correction such that
            pk_corrected = pk_target * pk_correction
        '''

        if wavemode is None:
            raise ValueError('Please specify a wavemode vector')

        from scipy.special import erf, erfc

        # this k could be truly k_nonlinear (in this case this must be computed
        # like in compute_halofit) or another sensible k
        knl = run['targetcosm'].get_nonlinear_scale()

        a_in_target = run['targetcosm'].expfactor
        a_in_scaled = run['scaledcosm'].expfactor
        if nlin:
            pk_lin_target = run['targetcosm'].get_nlpowerspec_z(
                expfactor=a_in_target, wavemode=wavemode, cold=cold)
            pk_lin_scaled = run['scaledcosm'].get_nlpowerspec_z(
                expfactor=a_in_scaled, wavemode=wavemode, cold=cold)

        else:
            pk_lin_target = run['targetcosm'].get_powerspec_z(
                expfactor=a_in_target, wavemode=wavemode, cold=cold)
            pk_lin_scaled = run['scaledcosm'].get_powerspec_z(
                expfactor=a_in_scaled, wavemode=wavemode, cold=cold)

        if correction_type == 'multiplicative':
            pk_correction = (pk_lin_target / pk_lin_scaled) * erfc(wavemode / knl) + erf(wavemode / knl)
        elif correction_type == 'additive':
            pk_correction = (pk_lin_target - pk_lin_scaled) * 0.5*erfc(np.log10(wavemode / knl)/0.01)
        elif correction_type == 'additive_field':
            pk_correction = (np.sqrt(pk_lin_target) - np.sqrt(pk_lin_scaled)) * 0.5*erfc(np.log10(wavemode / knl)/0.01)
            # pk_correction = (pk_lin_target + pk_lin_scaled - 2.0 * np.sqrt(pk_lin_scaled * pk_lin_target)) * erfc(wavemode / knl)
        else:
            raise RuntimeError('correction type {} not understood'.format(correction_type))

        return pk_correction



[docs]    def estimate_errors(self, run, pk=True, concentration=True, sigma=True, da=True, mf=True, lss_correction=True):
        """
        This function estimates the magnitude of the error of the rescaled
        simulation for several quantities
        """

        errors = dict()

        if hasattr(run, 'scaledcosm') is False:
            run['scaledcosm'] = deepcopy(self.OriginalCosmology)
            run['scaledcosm'].set_rescaling(run['params'])

        # concentration-mass relation
        if concentration is True:
            mass = np.logspace(np.log10(4*self.MinMass), np.log10(0.25*self.MaxMass), num=10)
            dc = run['scaledcosm'].get_concentration(mass)/run['targetcosm'].get_concentration(mass)
            errors['conc'] = 100.*np.mean(np.abs(dc-1))

        # Nonlinear P(k)
        if pk is True:

            wave1 = np.linspace(0.45, 0.55, num=10)
            wave2 = np.linspace(0.95, 1.05, num=10)
            wave3 = np.linspace(1.95, 2.05, num=10)

            if lss_correction == True:
                pkc1 = self.pk_correction(run, wave1, correction_type='additive',cold=True)
                pkc2 = self.pk_correction(run, wave2, correction_type='additive',cold=True)
                pkc3 = self.pk_correction(run, wave3, correction_type='additive',cold=True)
            else:
                pkc1 = 0
                pkc2 = 0
                pkc3 = 0

            dk1 = (pkc1 + run['scaledcosm'].get_nlpowerspec(
                wave1, cold=True))/run['targetcosm'].get_nlpowerspec(wave1, cold=True)
            dk2 = (pkc2 + run['scaledcosm'].get_nlpowerspec(
                wave2, cold=True))/run['targetcosm'].get_nlpowerspec(wave2, cold=True)
            dk3 = (pkc3 + run['scaledcosm'].get_nlpowerspec(
                wave3, cold=True))/run['targetcosm'].get_nlpowerspec(wave3, cold=True)

            errors['pk_05'] = 100.*np.mean(np.abs(dk1-1))
            errors['pk_10'] = 100.*np.mean(np.abs(dk2-1))
            errors['pk_20'] = 100.*np.mean(np.abs(dk3-1))

        # Growth history
        if da is True:
            errors['da'] = self.compute_growthdiff(run['params']['scaleTime'],
                                                   run['params']['scaleLength'],
                                                   run['targetcosm'])

        # Variance
        if sigma is True:
            radius = np.logspace(np.log10(0.3), np.log10(10), num=25)
            errors['sigma'] = self.compute_sigmadiff(run['params']['scaleTime'],
                                                     run['params']['scaleLength'],
                                                     run['targetcosm'],
                                                     radius)

        # Mass function
        if mf is True:
            mass = np.logspace(8, 15, 100)
            dmf = run['scaledcosm'].compute_massfunction(
                mass=mass)/run['targetcosm'].compute_massfunction(mass=mass)
            errors['mf'] = 100.*np.mean(np.abs(dmf-1))

        return errors

[docs]    def cost_function(self, pars):
        """
        This defines the function to minimise to find the best scaling parameter
        :param pars: The first and secdon elements are the length and time scaling factors, respectively.

        """

        scaleLength, scaleTime = pars
        if self.MinGrowth <= scaleTime <= self.MaxGrowth and 0.5 <= scaleLength <= 2.0:
            diff_sigma, diff_conc = self._scaled_statistics(scaleLength, scaleTime, verbose=False)
            fitIndex_radius = (self.ScaledCosmology.Variance['radius'] > self.MinRadius) & (
                self.ScaledCosmology.Variance['radius'] < self.MaxRadius)
            fitIndex_mass = (self.ScaledCosmology.Variance['mass'] > self.MinMass) & (
                self.ScaledCosmology.Variance['mass'] < self.MaxMass)
            loglik = -0.5 * \
                (0.9*np.mean(diff_sigma[fitIndex_radius]**2) +
                 0.1*np.mean(diff_conc[fitIndex_mass]**2))/0.01
            return loglik
        else:
            return -np.inf

        """
        Here we find the values of the transformation that minimise the function defined in cost_function()

        try:
            import emcee

            raise ImportError("Try installing emcee to do a MCMC fitting...")

        start = time.time()
        if TargetCosmology.__class__.__name__ != "Cosmology":
            raise ValueError("Target cosmology is not of class 'Cosmology'")
        else:
            self.TargetCosmology = TargetCosmology
            self.targetRedshift = 1./TargetCosmology.expfactor - 1

        # Choose an initial set of positions for the walkers.
        pos0 = [(np.random.rand(self.ndim)+0.5) for i in xrange(self.nwalkers)]

        # Initialize the sampler with the chosen specs.
        sampler = emcee.EnsembleSampler(self.nwalkers, self.ndim, self.cost_function)

        sampler.run_mcmc(pos0, 1000)

        samples = sampler.chain[:, 100:, :].reshape((-1, self.ndim))
        self.samples = samples

        import corner
        # ,range=[(0.5,1.5), (0.2,1.2)])
        fig = corner.corner(self.samples, labels=["Scale Length", "Scale Time"], smooth=2, bins=50)
        fig.show()

        scaleLength, scaleTime = map(lambda v: (
            v[1], v[2]-v[1], v[1]-v[0]), zip(*np.percentile(samples, [16, 50, 84], axis=0)))
        self._scaled_statistics(scaleLength[0], scaleTime[0])

        end = time.time()
        if self.verbose:
            logger.info("...done in %5.3g s" % (end - start))

        return scaleTime, scaleLength
