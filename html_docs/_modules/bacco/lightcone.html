    """
    A class that build a lightcone starting from a N-body simulation.
    """
    def __init__(self, bdir=None, zmax=2., field=[5,5], dm_file=None, halo_file=None, scaled_sim=False, verbose=None, **scaling_kwargs):
        """
        bdir:  base directory of the Nbody simulation.
        zmax: maximum redshift up to which the lightcone is built.
        field: field of view of the lightcone in degrees.
        scaled_sim: whether or not to scale the simulation.
        scaling_kwargs: kwargs to pass to the class ScaledSimulation.
        """

        self.bdir = bdir
        self.zmax = zmax
        self.field = np.array([field[0]*np.pi/180,field[1]*np.pi/180])

        sim = ScaledSimulation(basedir=self.bdir, dm_file=dm_file, halo_file=halo_file, **scaling_kwargs) if scaled_sim else Simulation(basedir=self.bdir, dm_file=dm_file, halo_file=halo_file)
        snaplist = deepcopy(sim.snaplist)
        self.Cosmology = sim.Cosmology
        boxsize = sim.header['BoxSize']
        del sim
        self.verbose = verbose if verbose is not None else config['verbose']

        self.m = (field[0]*180/(field[1]**2*np.pi))**(1/3)
        self.n = 180/(self.m**2*field[1]*np.pi)

        size = 1/(self.m**2*self.n)*1/(self.n**2*self.m)*(180/np.pi)**2 #square degrees
        d_first_replica = self.m*self.n*boxsize
        z_box = self.Cosmology.redshift_to_comoving_distance(boxsize)
        z_first_replica = self.Cosmology.redshift_to_comoving_distance(d_first_replica)

        self.origin = np.array([boxsize/2,boxsize/2,0])
        self.los = np.array([self.n, self.m, self.m*self.m])
        snap0 = find_closest_snaps_to(snaplist, 1)[0][0]
        snaplist['comoving_dist'] = [self.Cosmology.comoving_radial_distance(z)[0] for z in snaplist['z']]

        #check if the comoving distance between the snapshot is ever larger than the boxsize
        dz = np.abs(np.diff(snaplist['comoving_dist'][snap0::-1]))
        boxsize = 52
        if np.any(dz > boxsize):
            zm = snaplist['z'][snap0:0:-1][dz > boxsize][0]
            if zm < self.zmax:
                raise ValueError("Snapshots from z={} too sparse to create a continuous lightcone without box replications".format(zm))
        import re
        regex = re.compile(r'_\d+')

        self.particle_pos = []
        for i,snap in enumerate(np.arange(snap0+1)[::-1]):
            dm_file = regex.sub('_{0:03d}'.format(snap), dm_file) if dm_file is not None else None
            halo_file = regex.sub('_{0:03d}'.format(snap), halo_file) if halo_file is not None else None
            if scaled_sim:
                scaling_kwargs['cosmology'].set_expfactor(snaplist['a'][snap])
            sim = ScaledSimulation(basedir=self.bdir, dm_file=dm_file, halo_file=halo_file, **scaling_kwargs) if scaled_sim else Simulation(basedir=self.bdir, dm_file=dm_file, halo_file=halo_file)

            pos = sim.sdm['pos'] if dm_file is None else sim.dm['pos']
            pos -= (self.origin-(0,0,snaplist['comoving_dist'][snap])) #adding distances of different snapshots
            distances_from_origin = np.sqrt(np.sum(pos**2, axis=1))
            tan_alpha, tan_phi = pos[:,0]/pos[:,2], pos[:,1]/pos[:,2]
            #condition 1: force to use the particles of the snapshot closer to a given comoving distance from the origin
            condition1 = (np.abs(snaplist['comoving_dist'][snap]-distances_from_origin) < np.abs(snaplist['comoving_dist'][snap+1]-distances_from_origin) )
            #condition 2: pick all the objects that fall in the given field of view.
            condition2 = (np.abs(tan_alpha) <= np.tan(self.field[0]/2)) & (np.abs(tan_phi) <= np.tan(self.field[1]/2))
            mask = condition1 & condition2
            self.particle_pos.append(pos[mask])
            if self.verbose:
                print ("Iteration {}: Using Snapshot {};\n z={:.2f}; D_c={:.2f} Mpc/h.".format(i,snap,snaplist['z'][snap], snaplist['comoving_dist'][snap]))
                print ("Number of snapshot objects in the lightcone: {}".format(np.sum(mask)))

            if snaplist['z'][snap] >= self.zmax: break

        self.particle_pos = np.concatenate(self.particle_pos)

    def __str__(self):
        descr = ('Light cone of simulation {} \n' +
                'built up to z=: {} \n').format(self.bdir, self.zmax)
        return descr

    def vis_lightcone(self, filename=None, npix=(256,256), **kwargs):
        if not kwargs.keys():
                Xmax=max(self.particle_pos[:,0])
                Ymax=max(self.particle_pos[:,1])
                Zmax=max(self.particle_pos[:,2])/2
        if 'eye' not in kwargs.keys():
                kwargs['eye'] = (Xmax,Ymax,0)
        if 'lookat' not in kwargs.keys():
                kwargs['lookat'] = (-Xmax,-Ymax,Zmax)
        if 'width' not in kwargs.keys():
                kwargs['width'] = Zmax
        if 'zclip' not in kwargs.keys():
                kwargs['zclip'] = (0, Zmax)

        from . visualization import Visualization, show_image
        vis = Visualization(self.particle_pos, npix=npix, boxsize=0)
        vis.set_ortho_projection(**kwargs)
        vis.set_smoothing(rbins=np.linspace(0,100,20, dtype=np.float32), rfac=1e-3, extsize=10)
        rho = vis.image(show=False, usecic=False)
        show_image(rho+np.mean(rho)*1e-3, filename=filename, colormap="cubehelix")

    def plot_lightcone(self):
        import matplotlib.pyplot as plt
        import mpl_toolkits.mplot3d.axes3d as p3
        fig = plt.figure()
        ax = p3.Axes3D(fig)
        X=self.particle_pos[:,0]; Y=self.particle_pos[:,1]; Z=self.particle_pos[:,2]
        ax.scatter(X,Y,Z,s=1, marker=',', c='k')

        equal=True
        if equal:
            ax.set_aspect('equal')
            max_range = np.array([X.max()-X.min(), Y.max()-Y.min(), Z.max()-Z.min()]).max() / 2.0
            mid_x = (X.max()+X.min()) * 0.5
            mid_y = (Y.max()+Y.min()) * 0.5
            mid_z = (Z.max()+Z.min()) * 0.5
            ax.set_xlim(mid_x - max_range, mid_x + max_range)
            ax.set_ylim(mid_y - max_range, mid_y + max_range)
            ax.set_zlim(mid_z - max_range, mid_z + max_range)

        plt.show()
